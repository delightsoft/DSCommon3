processLevel = (result, map) ->

  name = undefined

  result.context ((path) -> (Result.prop name) path), ->

    map.$$list.forEach (field) ->

      name = field.name

      compileType result, field.$$src, field, context: 'field'

      processLevel result, field.fields if field.fields

      return
