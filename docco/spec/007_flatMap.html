    Флетмап добавляет в коллекцию sortedMap, коллекцию $$flat, которая содержит элементы всех уровней. В качестве ключей в $$flat используются полные имена элементов, и есть полный список элементов, в порядке как они приходтся при обходе структуры, в $$flat.$$list.

        @flatMap =
      item1: @item1 = {name: 'item1'}
      item2: @item2 =
        name: 'item2'
        subitems:
          item2a: @item2a = {name: 'item2a'}
          item2b: @item2b = {name: 'item2b'}
          $$list: [@item2a, @item2b]
      item3: @item3 = {name: 'item3'}
      $$list: [@item1, @item2, @item3]
      $$flat:
        'item1': @item1
        'item2': @item2
        'item2.item2a': @item2a
        'item2.item2b': @item2b
        'item3': @item3
        $$list: [@item1, @item2, @item2a, @item2b, @item3]
